\newcommand{\plot}[3]
	{
		\begin{tikzpicture}
		\begin{axis}[
			title=#1,
			xlabel={Cluster size},
			ylabel={Cut size},
			xmode=#3,
			legend entries={Structural sampler,Metis},
			legend style={
				anchor=north,
				at={(axis description cs:0,-0.1)}}
			]
		\addplot table [x=Input, y=StructuralSampler, col sep=comma] {#2};
		\addplot table [x=Input, y=Metis, col sep=comma] {#2};
		\end{axis}
		\end{tikzpicture}
	}
